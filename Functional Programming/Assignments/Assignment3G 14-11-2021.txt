using System;

namespace Assignment3G
{
    class Q1
    {
        static int count = 0;
        static Func<long, int> findNumberOfZeroes = (number) => {
            if(number > 0)
            {
                if (number % 10 == 0) count++;
                findNumberOfZeroes(number /= 10);
            }
            return count;
        };
        static void Main(string[] args)
        {
            Console.WriteLine(findNumberOfZeroes(100023));  // 3
            count = 0; Console.WriteLine(findNumberOfZeroes(120)); // 1
        }
    }
}

using System;
using System.Linq;

namespace Assignment3G
{
    class Q2
    {
        static Func<int, int, bool> isPrimeNumber = (number1, number2) => {
            if(number2 == 1) { return true; } 
            else { 
                if (number1 % number2 == 0) return false; 
                else { return isPrimeNumber(number1, number2 - 1); 
                } 
            }
        };
        static void Main(string[] args)
        {
            int[] primeNumbers = Enumerable.Range(3, 100-3).Where(x => isPrimeNumber(x, x / 2)).ToArray();
            Console.WriteLine("Prime numbers in range 3 to 100 are:" + string.Join(",", primeNumbers));
        }
    }
}

using LaYumba.Functional;
using System;
using System.Collections.Generic;
using static LaYumba.Functional.F;

namespace Assignment3G
{
    public static class Q3
    {
        public static Func<IEnumerable<int>, Func<int, bool>, Either<string, int>> findFirstOddNumber = (list, predicateFunc) => {
            foreach (int number in list)
                if (predicateFunc(number)) return number;
            return "Unable to find odd number in the enumerable list given";
        };

        static void Main(string[] args)
        {
            Func<int, bool> predicateOdd = i => i % 2 == 1;
            Console.WriteLine(findFirstOddNumber(new List<int> { 2, 4, 8 }, predicateOdd)); // Left(Unable to find odd number in the enumerable list given)
            Console.WriteLine(findFirstOddNumber(new List<int> { 2, 5, 7, 11 }, predicateOdd)); // Right(5)
        }
    }
}
