using System;
using System.Linq;

namespace functionalProgrammingTest
{
    class Program
    {
        static void Main(string[] args)
        {
            Tuple<double, double, double> circle;
            Console.WriteLine("Please enter the radius of the circle:");
            double radius = Convert.ToDouble(Console.ReadLine());
            circle = circleTuple(radius);
            Console.WriteLine("Area of the circle:" + circle.Item1);
            Console.WriteLine("Perimeter of the circle:" + circle.Item2);
            Console.WriteLine("Diameter of the circle:" + circle.Item3);
        }

        static Tuple<double, double, double> circleTuple(double radius)
        {
            double perimeter = 2 * Math.PI * radius;
            double diameter = 2 * radius;
            double area = Math.PI * radius * radius;
            return Tuple.Create(area, perimeter, diameter);
        }
    }
}

using System;

            Console.WriteLine("Area:" + tupleForCircle.Item1 + ", Perimeter:" + tupleForCircle.Item2 + ", Diameter:" + tupleForCircle.Item3);
        }

        static void calculateCircleParameters(double radiusOfCircle, out double areaOfCircle, out double perimeterOfCircle, out double diameterOfCircle)
        {
            Console.WriteLine("Given radius of the cirlce:" + radiusOfCircle);
            Console.WriteLine("Calculating area of circle:");
            areaOfCircle = Math.PI * radiusOfCircle * radiusOfCircle;
            Console.WriteLine("Area of the circle calculated:" + areaOfCircle);
            Console.WriteLine("Calculating perimeter of circle:");
            perimeterOfCircle = 2 * Math.PI * radiusOfCircle;
            Console.WriteLine("Perimeter of the circle calculated:" + perimeterOfCircle);
            Console.WriteLine("Calculating diameter of circle:");
            diameterOfCircle = 2 * radiusOfCircle;
            Console.WriteLine("Diameter of the circle calculated:" + diameterOfCircle);
        }
    }

using System;

namespace ConsoleApplicationTest 
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Enter radius of circle:"); double r = Convert.ToDouble(Console.ReadLine());
            calculateAreaPerDia(r, out double a, out double p, out double d);
            Tuple<double, double, double>  t = Tuple.Create(a, p, d);
            Console.WriteLine("Radius:" + r + " -----> Area:" + t.Item1 + ", Perimeter:" + t.Item2 + ", Diameter:" + t.Item3);
        }

        static void calculateAreaPerDia(double r, out double a, out double p, out double d)
        {
            a = Math.PI * r * r; p = 2 * Math.PI * r; d = 2 * r;
        }
    }
}
